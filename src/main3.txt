#include <Arduino.h>
#include <Wire.h>
#include <SPI.h>
#include <lvgl.h>
#include "M5Dial.h"
#include "ui.h"


static lv_display_t *disp;
static lv_indev_t *indev;

#define LV_VER_RES_MAX 240
#define LV_HOR_RES_MAX 240

// --------------------
// LVGL flush callback
// --------------------
void my_disp_flush(lv_display_t *d, const lv_area_t *area, uint8_t *px_map) {
    M5Dial.Display.startWrite();
    M5Dial.Display.pushImageDMA(
        area->x1, area->y1,
        lv_area_get_width(area),
        lv_area_get_height(area),
        (lgfx::rgb565_t *)px_map
    );
    M5Dial.Display.endWrite();
    lv_display_flush_ready(d); // Let LVGL know we're done
}

// --------------------
// LVGL touch callback
// --------------------
void my_touch_read(lv_indev_t *drv, lv_indev_data_t *data) {
    auto t = M5Dial.Touch.getDetail();
    if (t.isPressed()) {
        data->state = LV_INDEV_STATE_PRESSED;
        data->point.x = t.x;
        data->point.y = t.y;
    } else {
        data->state = LV_INDEV_STATE_RELEASED;
    }
}


// --------------------
// LVGL display init
// --------------------
void tft_lv_initialization() {
    lv_init();

    // Allocate full-screen DMA buffer
    void *buf1 = heap_caps_malloc(
        LV_HOR_RES_MAX * LV_VER_RES_MAX * sizeof(lv_color_t),
        MALLOC_CAP_DMA
    );

    disp = lv_display_create(LV_HOR_RES_MAX, LV_VER_RES_MAX);
    lv_display_set_flush_cb(disp, my_disp_flush);
    lv_display_set_buffers(
        disp,
        buf1, NULL,
        LV_HOR_RES_MAX * LV_VER_RES_MAX * sizeof(lv_color_t),
        LV_DISPLAY_RENDER_MODE_FULL
    );

    indev = lv_indev_create();
    lv_indev_set_type(indev, LV_INDEV_TYPE_POINTER);
    lv_indev_set_read_cb(indev, my_touch_read);
}


void setup() {
    auto cfg = M5.config();
    M5Dial.begin(cfg, true, false);
    M5Dial.Display.setBrightness(100);
    Serial.begin(115200);

    tft_lv_initialization();

    ui_init();
    
    Serial.begin(115200);
}


void loop() {
  uint32_t wait_ms = lv_timer_handler();
  lv_tick_inc(wait_ms);
  M5.delay(wait_ms);

  M5Dial.update();

}
